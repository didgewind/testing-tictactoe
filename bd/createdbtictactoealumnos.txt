# Crea 15 bases de datos tictactoe_alumnoX con usuario y password alumnoX

### alumno1 ###

drop database if exists tictactoe_alumno1;

create database if not exists tictactoe_alumno1;
use tictactoe_alumno1;

## CREATES ##

# Tablas de entidades

CREATE TABLE IF NOT EXISTS movimientos 
(numMovimiento INTEGER, 
x INTEGER,
y INTEGER, 
player CHAR, 
primary key (numMovimiento));

grant all on tictactoe_alumno1.* to alumno1@'%' identified by 'alumno1';
